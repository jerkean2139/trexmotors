  <script>
    // Gallery functionality
    document.addEventListener('DOMContentLoaded', function() {
      const mainImages = document.querySelectorAll('.main-image img');
      const dots = document.querySelectorAll('.gallery-dot');
      const thumbnails = document.querySelectorAll('.thumbnail');
      const prevButton = document.querySelector('.gallery-control.prev');
      const nextButton = document.querySelector('.gallery-control.next');
      let currentIndex = 0;
      
      // Function to update the active image
      function updateActiveImage(index) {
        mainImages.forEach(img => img.classList.remove('active'));
        dots.forEach(dot => dot.classList.remove('active'));
        thumbnails.forEach(thumb => thumb.classList.remove('active'));
        
        mainImages[index].classList.add('active');
        dots[index].classList.add('active');
        thumbnails[index].classList.add('active');
        currentIndex = index;
      }
      
      // Event listeners for dots
      dots.forEach((dot, index) => {
        dot.addEventListener('click', () => {
          updateActiveImage(index);
        });
      });
      
      // Event listeners for thumbnails
      thumbnails.forEach((thumbnail, index) => {
        thumbnail.addEventListener('click', () => {
          updateActiveImage(index);
        });
      });
      
      // Event listeners for prev/next buttons
      prevButton.addEventListener('click', () => {
        let newIndex = currentIndex - 1;
        if (newIndex < 0) newIndex = mainImages.length - 1;
        updateActiveImage(newIndex);
      });
      
      nextButton.addEventListener('click', () => {
        let newIndex = currentIndex + 1;
        if (newIndex >= mainImages.length) newIndex = 0;
        updateActiveImage(newIndex);
      });
      
      // Auto-rotate images every 5 seconds
      setInterval(() => {
        let newIndex = currentIndex + 1;
        if (newIndex >= mainImages.length) newIndex = 0;
        updateActiveImage(newIndex);
      }, 5000);
    });
  </script>
